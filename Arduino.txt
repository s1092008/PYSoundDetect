#define Relay 8
#define BL 3
#define RL 4
#define GL 5
String str;
void setup() {
  pinMode(Relay, OUTPUT);
  Serial.begin(9600);
  digitalWrite(Relay, HIGH);
}
void loop() {
  if (Serial.available()) {
    str = Serial.readStringUntil('\n');
 
    if (str == "go") {          
        digitalWrite(Relay, LOW);     // 通電
        delay(5000);
        digitalWrite(BL,HIGH);
        digitalWrite(Relay, HIGH);
    } else if (str == "stop") {
        digitalWrite(Relay, HIGH);
        digitalWrite(RL,HIGH);
    }
  }
  else{
    digitalWrite(GL.HIGH);
  }
}
