#define Relay 9
String str;
void setup() {
  pinMode(Relay, OUTPUT);
  Serial.begin(9600);
  digitalWrite(Relay, HIGH);
}
void loop() {
  if (Serial.available()) {
    str = Serial.readStringUntil('\n');
    Serial.println(str);  
    if (str == "go") {             
      digitalWrite(Relay, LOW);
      delay(3000);
      digitalWrite(Relay, HIGH);
    } 
  }
}
